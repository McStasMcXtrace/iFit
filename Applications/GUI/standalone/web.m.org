function web(varargin)

if length(varargin)
  url = varargin{1};
else
  url = '';
end
if ~isdeployed && usejava('jvm')
  builtin('web',varargin{:});
else
  disp([ mfilename ' ' url ]);
  % attempts to guess how to launch the browser
  if ispc
    system([ 'start /D "' ifitpath '" Docs']);
  elseif ismac
    system([ 'open "' url '"' ]);
  else
    browser = getenv('BROWSER');
    if isempty(browser)
      [flag, out]=system('locate firefox');
      if isempty(editor) && ~isempty(out), editor='firefox'; end
      [flag, out]=system('locate google-chrome');
      if isempty(editor) && ~isempty(out), editor='google-chrome'; end
      [flag, out]=system('locate konqueror');
      if isempty(editor) && ~isempty(out), editor='konqueror'; end
      [flag, out]=system('locate seamonkey');
      if isempty(editor) && ~isempty(out), editor='seamonkey'; end
    end
    if isempty(browser), browser = 'gnome-open'; end
    if ~isempty(url)
      browser = [ browser ' "' url '"' ];
    end
    system( browser );
  end
end
